          "listeners": [ {
            "event": "init",
            "method": function( e ) {
              var chart = e.chart;

              function getCurrentData() {
                var data = chartData[currentYear];
                currentYear++;
                if (currentYear > 2014)
                  currentYear = 1995;
                return data;
              }

              function loop() {
                chart.allLabels[0].text = currentYear;
                var data = getCurrentData();
                chart.animateData( data, {
                  duration: 1000,
                  complete: function() {
                    setTimeout( loop, 3000 );
                  }
                } );
              }

              loop();
            }
          } ],
           "export": {
           "enabled": true
          }
        } );
    });
</script>
